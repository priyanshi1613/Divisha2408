Imports System.Data.OleDb
Partial Public Class crud
    Inherits System.Web.UI.Page
    Dim cn As New OleDbConnection("Provider=Microsoft.ACE.OLEDB.12.0;Data Source=C:\Users\HP\Documents\nidhi.accdb")
    
    Protected Sub btninsert_Click(ByVal sender As Object, ByVal e As EventArgs) Handles btninsert.Click
        Try
            Dim str As String
            Dim ans As Integer
            str = "insert into student values(" & txtrno.Text & ",'" & txtname.Text & "')"
            cn.Open()
            Dim cmd As New OleDbCommand(str, cn)
            ans = cmd.ExecuteNonQuery
            MsgBox("inserted successfully")
            GridView1.DataBind()
            cn.Close()
        Catch ex As Exception
            MsgBox(ex.ToString)
        End Try
    End Sub

    Protected Sub btnupdate_Click(ByVal sender As Object, ByVal e As EventArgs) Handles btnupdate.Click
        Try
            Dim str As String
            Dim ans As Integer
            str = "update student set name='" & txtname.Text & "' where id=" & txtrno.Text & ""
            cn.Open()
            Dim cmd As New OleDbCommand(str, cn)
            ans = cmd.ExecuteNonQuery
            MsgBox("update successfully")
            GridView1.DataBind()
            cn.Close()
        Catch ex As Exception
            MsgBox(ex.ToString)
        End Try
    End Sub

    Protected Sub btndelete_Click(ByVal sender As Object, ByVal e As EventArgs) Handles btndelete.Click
        Try
            Dim str As String
            Dim ans As Integer
            str = "delete from student where id=" & txtrno.Text & ""
            cn.Open()
            Dim cmd As New OleDbCommand(str, cn)
            ans = cmd.ExecuteNonQuery
            MsgBox("delete successfully")
            GridView1.DataBind()
            cn.Close()
        Catch ex As Exception
            MsgBox(ex.ToString)
        End Try
    End Sub

    Protected Sub btnsearch_Click(ByVal sender As Object, ByVal e As EventArgs) Handles btnsearch.Click
        Try
            Dim str As String
            str = "select * from student where id=" & txtrno.Text & ""
            cn.Open()
            Dim cmd As New OleDbCommand(str, cn)
            Dim dr As OleDbDataReader
            dr = cmd.ExecuteReader
            While dr.Read
                txtname.Text = dr.Item(1).ToString
            End While
            GridView1.DataBind()
            cn.Close()
        Catch ex As Exception
            MsgBox(ex.ToString)
        End Try
    End Sub
End Class
